   [semantics_def]  Definition
      
      ⊢ ∀s start.
          semantics s start =
          (let
             prog = Call NONE (SOME start) [] NONE
           in
             if
               ∃k. case FST (evaluate (prog,s with clock := k)) of
                     NONE => T
                   | SOME (Result v5) => F
                   | SOME (Exception v6) => T
                   | SOME Break => T
                   | SOME Continue => T
                   | SOME TimeOut => F
                   | SOME (FinalFFI v7) => F
                   | SOME Error => T

               Fail

               case
                 some res. ∃k t r outcome.
                   evaluate (prog,s with clock := k) = (r,t) ∧
                   (case r of
                      NONE => F
                    | SOME (Result v5) => outcome = Success
                    | SOME (Exception v6) => F
                    | SOME Break => F
                    | SOME Continue => F
                    | SOME TimeOut => F
                    | SOME (FinalFFI e) => outcome = FFI_outcome e
                    | SOME Error => F) ∧
                   res = Terminate outcome t.ffi.io_events
               of
                 NONE =>
                   Diverge
                     (LUB
                        (IMAGE
                           (λk.
                                fromList
                                  (SND (evaluate (prog,s with clock := k))).

               | SOME res => res)
