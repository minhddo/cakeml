   [comp_ffi_def]  Definition
      
      ⊢ ∀fs t.
          comp_ffi fs t =
          WHILE
            (λ(t,st). case t of Ret v4 => F | Tau v5 => T | Vis v6 v7 => T)
            (λ(t,st).
                 case t of
                   Ret v => (t,st)
                 | Tau v1 => (v1,st)
                 | Vis (s,c,ws) g =>
                   case FST fs s st c ws of
                     Oracle_return fs' ws' =>
                       if LENGTH ws = LENGTH ws' then
                         (g (INL (INR ws')),fs')
                       else (g (INL (INL FFI_failed)),st)
                   | Oracle_final outcome => (g (INL (INL outcome)),st))
            (t,SND fs)

   [h_prog_def_primitive]  Definition
      
      ⊢ h_prog =
        WFREC (@R. WF R)
          (λh_prog a.
               case a of
                 (Skip,s) => I (Ret (INR (NONE,s)))
               | (Dec vname sh e p,s) => I (h_prog_dec vname sh e p s)
               | (Assign vk vname' e',s) =>
                 I (h_prog_assign vk vname' e' s)
               | (Primitive vname'' pop es,s) =>
                 I (h_prog_primitive vname'' pop es s)
               | (Store dst src,s) => I (h_prog_store dst src s)
               | (Store32 dst' src',s) => I (h_prog_store_32 dst' src' s)
               | (StoreByte dst'' src'',s) =>
                 I (h_prog_store_byte dst'' src'' s)
               | (Seq p1 p2,s) => I (h_prog_seq p1 p2 s)
               | (If gexp p1' p2',s) => I (h_prog_cond gexp p1' p2' s)
               | (While gexp' p',s) => I (h_prog_while gexp' p' s)
               | (Break,s) => I (Ret (INR (SOME Break,s)))
               | (Continue,s) => I (Ret (INR (SOME Continue,s)))
               | (Call calltyp tgtexp argexps,s) =>
                 I (h_prog_call calltyp tgtexp argexps s)
               | (DecCall rt shape tgtexp' argexps' prog1,s) =>
                 I (h_prog_deccall rt shape tgtexp' argexps' prog1 s)
               | (ExtCall ffi_name conf_ptr conf_len array_ptr array_len,s) =>
                 I
                   (h_prog_ext_call ffi_name conf_ptr conf_len array_ptr
                      array_len s)
               | (Raise eid e'³',s) => I (h_prog_raise eid e'³' s)
               | (Return e'⁴',s) => I (h_prog_return e'⁴' s)
               | (ShMemLoad op vk' v ad,s) =>
                 I (h_prog_sh_mem_load op vk' v ad s)
               | (ShMemStore op' ad' e'',s) =>
                 I (h_prog_sh_mem_store op' ad' e'' s)
               | (Tick,s) => I (Ret (INR (NONE,s)))
               | (Annot v0 v1,s) => I (Ret (INR (NONE,s))))

            Ret (INR (SOME Error,s))

           SOME (ValWord array_ptr_adr),SOME (ValWord array_sz)) =>
            (case
               (read_bytearray conf_ptr_adr (w2n conf_sz)
                  (mem_load_byte s.memory s.memaddrs s.be),

                                         new_bytes s.memory s.memaddrs s.be
                                   in
                                     (NONE,s with memory := nmem))

                                     (FinalFFI
                                        (Final_event (ExtCall ffi_name)
                                           conf_bytes array_bytes
                                           FFI_failed)),empty_locals s)
                              | INR v1 => (SOME Error,s))))
               else
                 Ret
                   (INR
                      (NONE,
                       s with
                       memory :=
                         write_bytearray array_ptr_adr array_bytes s.memory
                           s.memaddrs s.be)))
          | (SOME (Val v14),SOME (Val v28),SOME (Val v42),
           SOME (RStruct v53)) =>
            Ret (INR (SOME Error,s))
          | (SOME (Val v14),SOME (Val v28),SOME (Val v42),
