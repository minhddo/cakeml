      ⊢ ∀args.
          crep_primop AddCarry args =
          if LENGTH args = 3 ∧ EVERY isWord args then
            (let
               l = theWord args❲0❳;
               r = theWord args❲1❳;

   [semantics_def]  Definition
      
      ⊢ ∀s start.
          semantics s start =
          (let
             prog = Call NONE start []
           in
             if
               ∃k. case FST (evaluate (prog,s with clock := k)) of
                     NONE => T
                   | SOME Error => T
                   | SOME TimeOut => F
                   | SOME Break => T
                   | SOME Continue => T
                   | SOME (Return v5) => F
                   | SOME (Exception v6) => T
                   | SOME (FinalFFI v7) => F
             then
               Fail
             else
               case
                 some res. ∃k t r outcome.
                   evaluate (prog,s with clock := k) = (r,t) ∧
                   (case r of
                      NONE => F
                    | SOME Error => F
                    | SOME TimeOut => F
                    | SOME Break => F
                    | SOME Continue => F
                    | SOME (Return v5) => outcome = Success
                    | SOME (Exception v6) => F
                    | SOME (FinalFFI e) => outcome = FFI_outcome e) ∧
                   res = Terminate outcome t.ffi.io_events
               of
                 NONE =>
                   Diverge
                     (LUB
                        (IMAGE
                           (λk.
                                fromList
                                  (SND (evaluate (prog,s with clock := k))).
                                  ffi.io_events) 𝕌(:num)))
               | SOME res => res)

          sh_mem_load v addr nb s =
          if nb = 0 then
            if addr ∈ s.sh_memaddrs then
              case
                call_FFI s.ffi (SharedMem MappedRead) [n2w nb]
                  (word_to_bytes addr F)
              of
                FFI_return new_ffi new_bytes =>
                  (NONE,
                   set_var v (Word (word_of_bytes F 0w new_bytes)) s with
                   ffi := new_ffi)

                (SOME (FinalFFI outcome),empty_locals s)

              call_FFI s.ffi (SharedMem MappedRead) [n2w nb]

        (∀v s. eval s (Var v) = FLOOKUP s.locals v) ∧
        (∀s addr.
           eval s (Load addr) =
           case eval s addr of NONE => NONE | SOME (Word w) => mem_load w s) ∧
        (∀s addr.
           eval s (Load32 addr) =
           case eval s addr of
             NONE => NONE
           | SOME (Word w) =>
             case mem_load_32 s.memory s.memaddrs s.be w of
               NONE => NONE
             | SOME w => SOME (Word (w2w w))) ∧
        (∀s addr.
           eval s (LoadByte addr) =
           case eval s addr of
             NONE => NONE
           | SOME (Word w) =>
             case mem_load_byte s.memory s.memaddrs s.be w of
               NONE => NONE
             | SOME w => SOME (Word (w2w w))) ∧
        (∀s gadr. eval s (LoadGlob gadr) = FLOOKUP s.globals gadr) ∧
        (∀s op es.
           eval s (Op op es) =
           case OPT_MMAP (λa. eval s a) es of
             NONE => NONE
           | SOME ws =>
             if EVERY (λw. T) ws then
               OPTION_MAP Word
                 (word_op op (MAP (λw. case w of Word n => n) ws))
             else NONE) ∧
        (∀s op es.
           eval s (Crepop op es) =
           case OPT_MMAP (λa. eval s a) es of
             NONE => NONE
           | SOME ws =>
             if EVERY (λw. T) ws then
               OPTION_MAP Word
                 (crep_op op (MAP (λw. case w of Word n => n) ws))
             else NONE) ∧
        (∀s e2 e1 cmp.
           eval s (Cmp cmp e1 e2) =
           case (eval s e1,eval s e2) of
             (NONE,v3) => NONE
           | (SOME v5,NONE) => NONE
           | (SOME (Word w1),SOME (Word w2)) =>
             SOME (Word (v2w [word_cmp cmp w1 w2]))) ∧
        (∀sh s n e.

             NONE => NONE
           | SOME (Word w) => OPTION_MAP Word (word_sh sh w n)) ∧

   [evaluate_def]  Theorem
      
      ⊢ (∀s. evaluate (Skip,s) = (NONE,s)) ∧
        (∀v s prog e.
           evaluate (Dec v e prog,s) =
           case eval s e of
             NONE => (SOME Error,s)
           | SOME value =>
             (let
                (res,st) =
                  evaluate (prog,s with locals := s.locals⟨v ↦ value⟩)
              in
                (res,
                 st with locals := res_var st.locals (v,FLOOKUP s.locals v)))) ∧
        (∀s rhss pop lhss.
           evaluate (Primitive lhss pop rhss,s) =
           case OPT_MMAP (FLOOKUP s.locals) rhss of
             NONE => (SOME Error,s)
           | SOME ws =>
             case crep_primop pop ws of
               NONE => (SOME Error,s)
             | SOME res_ws =>
               if
                 LENGTH lhss = LENGTH res_ws ∧
                 EVERY (λv. IS_SOME (FLOOKUP s.locals v)) lhss ∧
                 ALL_DISTINCT lhss
               then
                 (NONE,s with locals := s.locals |++ ZIP (lhss,res_ws))
               else (SOME Error,s)) ∧
        (∀v src s.
           evaluate (Assign v src,s) =
           case eval s src of
             NONE => (SOME Error,s)
           | SOME w =>
             case FLOOKUP s.locals v of
               NONE => (SOME Error,s)
             | SOME v2 => (NONE,s with locals := s.locals⟨v ↦ w⟩)) ∧
        (∀src s dst.
           evaluate (Store dst src,s) =
           case (eval s dst,eval s src) of
             (NONE,v3) => (SOME Error,s)
           | (SOME v5,NONE) => (SOME Error,s)
           | (SOME (Word adr),SOME w) =>
