   [semantics_def]  Definition
      
      ⊢ ∀s start.
          semantics s start =
          (let
             prog = TailCall start []
           in
             if
               ∃k. case FST (evaluate (prog,s with clock := k)) of
                     NONE => T
                   | SOME Error => T
                   | SOME TimeOut => F
                   | SOME Break => T
                   | SOME Continue => T
                   | SOME (Return v6) => F
                   | SOME (Exception v7 v8) => T
                   | SOME (FinalFFI v9) => F
             then
               Fail
             else
               case
                 some res. ∃k t r outcome.
                   evaluate (prog,s with clock := k) = (r,t) ∧
                   (case r of
                      NONE => F
                    | SOME Error => F
                    | SOME TimeOut => F
                    | SOME Break => F
                    | SOME Continue => F
                    | SOME (Return v6) => outcome = Success
                    | SOME (Exception v7 v8) => F
                    | SOME (FinalFFI e) => outcome = FFI_outcome e) ∧
                   res = Terminate outcome t.ffi.io_events
               of
                 NONE =>
                   Diverge
                     (LUB
                        (IMAGE
                           (λk.
                                fromList
                                  (SND (evaluate (prog,s with clock := k))).
                                  ffi.io_events) 𝕌(:num)))
               | SOME res => res)
