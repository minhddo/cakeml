          (case x of
             Based => v0
           | NotBased => v1
           | Trusted => v2
           | NotTrusted => v3) =
          (λm.
               if m < 1 then v0
               else if m < 2 then v1

   [based_merge_def]  Definition
      
      ⊢ ∀x y.
          based_merge x y =
          case (x,y) of
            (Based,v1) => Based
          | (NotBased,Based) => Based
          | (NotBased,NotBased) => NotBased
          | (NotBased,Trusted) => Trusted
          | (NotBased,NotTrusted) => NotTrusted

          | (Trusted,NotBased) => Trusted
          | (Trusted,Trusted) => Trusted
          | (Trusted,NotTrusted) => NotTrusted
          | (NotTrusted,Based) => Based
          | (NotTrusted,NotBased) => NotTrusted
          | (NotTrusted,Trusted) => NotTrusted

      ⊢ ∀vctxt x y.
          branch_loc_inf vctxt x y =
          (let
             x' =
               mapWithKey
                 (λk v.
                      v with
                      vsh_bd :=
                        if ¬member k y then
                          case lookup vctxt k of
                            NONE => sh_bd_from_bd NotTrusted v.vsh_bd
                          | SOME v' => sh_bd_branch v.vsh_bd v'.vsh_bd
                        else v.vsh_bd) x;
             y' =
               mapWithKey
                 (λk v.
                      v with
                      vsh_bd :=
                        if ¬member k x then
                          case lookup vctxt k of
                            NONE => sh_bd_from_bd NotTrusted v.vsh_bd
                          | SOME v' => sh_bd_branch v.vsh_bd v'.vsh_bd
                        else v.vsh_bd) y

             unionWith
               (λvx vy. vx with vsh_bd := sh_bd_branch vx.vsh_bd vy.vsh_bd)
               x' y')

                 (ShapeErr
                    (get_non_word_msg «load variable»
                       (shape_to_str vinf.vshape) ctxt.loc ctxt.scope))
             else return ();
             return
               <|exits_fun := F; exits_loop := F; last := OtherLast;
                 var_delta := empty compare; curr_loc := ctxt.loc|>
           od) ∧
        (∀mop exp ctxt addr.
           static_check_prog ctxt (ShMemStore mop addr exp) =
           do
             aret <- static_check_exp ctxt addr;
             eret <- static_check_exp ctxt exp;
             case aret.sh_bd of
               WordB Based =>
                 log (WarningErr (get_memop_msg F F F ctxt.loc ctxt.scope))
             | WordB NotBased => return ()
             | WordB Trusted => return ()
             | WordB NotTrusted =>
               log (WarningErr (get_memop_msg F F T ctxt.loc ctxt.scope))
             | StructB v5 =>
               error
                 (ShapeErr
                    (get_non_word_msg «store address»
                       (sh_bd_to_str aret.sh_bd) ctxt.loc ctxt.scope))
             | NamedB v6 v7 =>
               error
                 (ShapeErr
                    (get_non_word_msg «store address»
                       (sh_bd_to_str aret.sh_bd) ctxt.loc ctxt.scope));
             if ¬sh_bd_has_shape One eret.sh_bd then
               error
                 (ShapeErr
                    (get_non_word_msg «store value»
                       (sh_bd_to_str eret.sh_bd) ctxt.loc ctxt.scope))
             else return ();
             return
               <|exits_fun := F; exits_loop := F; last := OtherLast;
                 var_delta := empty compare; curr_loc := ctxt.loc|>
           od) ∧
        (∀ctxt.
           static_check_prog ctxt Tick =
           return
             <|exits_fun := F; exits_loop := F; last := InvisLast;
               var_delta := empty compare; curr_loc := ctxt.loc|>) ∧
        ∀tag str ctxt.
          static_check_prog ctxt (Annot tag str) =
          (let
             loc =
               if tag = «location» then concat [«AT »; str; «: »]
               else ctxt.loc
           in
             return
               <|exits_fun := F; exits_loop := F; last := InvisLast;
                 var_delta := empty compare; curr_loc := loc|>)
   
   [static_check_prog_ind]  Theorem
      
      ⊢ ∀P. (∀ctxt. P ctxt Skip) ∧
            (∀ctxt vname shape exp prog.
               (∀ctxt' eret.
                  ctxt' =
                  ctxt with
                  <|locals :=
                      insert ctxt.locals vname <|vsh_bd := eret.sh_bd|>;
                    last := OtherLast|> ⇒
                  P ctxt' prog) ⇒
               P ctxt (Dec vname shape exp prog)) ∧
            (∀ctxt vname shape fname args prog.
               (∀ctxt' sb.
                  ctxt' =
                  ctxt with
                  <|locals := insert ctxt.locals vname <|vsh_bd := sb|>;
                    last := OtherLast|> ⇒
                  P ctxt' prog) ⇒
               P ctxt (DecCall vname shape fname args prog)) ∧
            (∀ctxt vname exp. P ctxt (Assign Local vname exp)) ∧
            (∀ctxt vname exp. P ctxt (Assign Global vname exp)) ∧
            (∀ctxt v pop es. P ctxt (Primitive v pop es)) ∧
            (∀ctxt vname hdl fname args.
               (∀v eid v2 evar prog evinf.
                  hdl = SOME v ∧ v = (eid,v2) ∧ v2 = (evar,prog) ⇒
                  P
                    (ctxt with
                     locals :=
                       insert ctxt.locals evar
                         (evinf with
                          vsh_bd := sh_bd_from_bd Trusted evinf.vsh_bd))
                    prog) ⇒
               P ctxt (AssignCall (Local,vname) hdl fname args)) ∧
            (∀ctxt vname hdl fname args.
               (∀v eid v2 evar prog evinf.
                  hdl = SOME v ∧ v = (eid,v2) ∧ v2 = (evar,prog) ⇒
                  P
                    (ctxt with
                     locals :=
                       insert ctxt.locals evar
                         (evinf with
                          vsh_bd := sh_bd_from_bd Trusted evinf.vsh_bd))
                    prog) ⇒
               P ctxt (AssignCall (Global,vname) hdl fname args)) ∧
            (∀ctxt exp. P ctxt (Return exp)) ∧
            (∀ctxt fname args. P ctxt (TailCall fname args)) ∧
            (∀ctxt hdl fname args.
               (∀v eid v2 evar prog evinf.
                  hdl = SOME v ∧ v = (eid,v2) ∧ v2 = (evar,prog) ⇒
                  P
                    (ctxt with
                     locals :=
                       insert ctxt.locals evar
                         (evinf with
                          vsh_bd := sh_bd_from_bd Trusted evinf.vsh_bd))
                    prog) ⇒
               P ctxt (StandAloneCall hdl fname args)) ∧
            (∀ctxt fname ptr1 len1 ptr2 len2.
               P ctxt (ExtCall fname ptr1 len1 ptr2 len2)) ∧
            (∀ctxt prog1 prog2.
               (∀warn_p1 ctxt1 next_r pret1 ctxt2 warn_p2 ctxt3.
                  (warn_p1,ctxt1) = reached_warnable prog1 ctxt ∧
                  next_r = next_is_reachable ctxt1.is_reachable pret1.last ∧
                  ctxt2 =
                  ctxt1 with
                  <|locals := seq_loc_inf ctxt1.locals pret1.var_delta;
                    is_reachable := next_r;
                    last :=
                      if
                        next_now_unreachable ctxt1.is_reachable next_r
                      then
                        pret1.last
                      else ctxt1.last; loc := pret1.curr_loc|> ∧
                  (warn_p2,ctxt3) = reached_warnable prog2 ctxt2 ⇒
                  P ctxt3 prog2) ∧
               (∀warn_p1 ctxt1.
                  (warn_p1,ctxt1) = reached_warnable prog1 ctxt ⇒
                  P ctxt1 prog1) ⇒
               P ctxt (Seq prog1 prog2)) ∧
            (∀ctxt exp prog1 prog2.
               (∀pret1. P (ctxt with loc := pret1.curr_loc) prog2) ∧
               P ctxt prog1 ⇒
               P ctxt (If exp prog1 prog2)) ∧
            (∀ctxt exp prog.
               P (ctxt with in_loop := T) prog ⇒ P ctxt (While exp prog)) ∧
            (∀ctxt. P ctxt Break) ∧ (∀ctxt. P ctxt Continue) ∧
            (∀ctxt eid exp. P ctxt (Raise eid exp)) ∧
            (∀ctxt addr exp. P ctxt (Store addr exp)) ∧
            (∀ctxt addr exp. P ctxt (Store32 addr exp)) ∧
            (∀ctxt addr exp. P ctxt (StoreByte addr exp)) ∧
            (∀ctxt mop vname addr. P ctxt (ShMemLoad mop Local vname addr)) ∧
            (∀ctxt mop vname addr. P ctxt (ShMemLoad mop Global vname addr)) ∧
            (∀ctxt mop addr exp. P ctxt (ShMemStore mop addr exp)) ∧
            (∀ctxt. P ctxt Tick) ∧ (∀ctxt tag str. P ctxt (Annot tag str)) ⇒
            ∀v v1. P v v1
   
   [static_check_progs_def]  Theorem
      
      ⊢ (∀sctxt gctxt fctxt.
           static_check_progs fctxt gctxt sctxt [] = return ()) ∧
        (∀v1 v0 sctxt gctxt fctxt decls.
           static_check_progs fctxt gctxt sctxt (Name v0 v1::decls) =
           static_check_progs fctxt gctxt sctxt decls) ∧
        (∀v4 v3 v2 sctxt gctxt fctxt decls.
           static_check_progs fctxt gctxt sctxt (Decl v2 v3 v4::decls) =
           static_check_progs fctxt gctxt sctxt decls) ∧
        ∀sctxt gctxt fi fctxt decls.
          static_check_progs fctxt gctxt sctxt (Function fi::decls) =
          (let
             (param_names,param_shapes) = UNZIP fi.params
           in
             do
               param_sbs <-
                 case
                   OPT_MMAP (sh_bd_from_sh sctxt Trusted) param_shapes
                 of
                   NONE =>
                     error
                       (ScopeErr
                          (get_implementation_err_msg
                             «static analysis failed to convert in-scope shape»
                             «» (FunScope fi.name «»)))
                 | SOME sbs => return (ZIP (param_names,sbs));
               ctxt <<-
                 <|locals :=
                     FOLDL (λm (v,sb). insert m v <|vsh_bd := sb|>)
                       (empty compare) param_sbs; globals := gctxt;
                   funcs := fctxt; structs := sctxt;
                   scope := FunScope fi.name «»; in_loop := F;
                   is_reachable := IsReach; last := InvisLast; loc := «»|>;
               prog_ret <- static_check_prog ctxt fi.body;
               if ¬prog_ret.exits_fun then
                 error
                   (GenErr
                      (concat
                         [«branches missing return statement in »;
                          get_scope_desc (FunScope fi.name «»); «\n»]))
               else return ();
               static_check_progs fctxt gctxt sctxt decls
             od)
   
   [static_check_progs_ind]  Theorem
      
      ⊢ ∀P. (∀fctxt gctxt sctxt. P fctxt gctxt sctxt []) ∧
            (∀fctxt gctxt sctxt v0 v1 decls.
               P fctxt gctxt sctxt decls ⇒

            ∀v v1 v2 v3. P v v1 v2 v3
