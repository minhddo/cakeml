      ⊢ ∀decs start.
          compile_top decs start =
          case ALOOKUP (functions decs) start of
            NONE => []
          | SOME (args,body) =>
            (let
               nds = resort_decls decs;
               start' = new_main_name decs;
               nds' = fperm_decs start start' nds;
               (decls,funs,ctxt) =
                 compile_decs
                   <|globals := FEMPTY; globals_size := 0w;
                     max_globals_size :=
                       bytes_in_word *
                       n2w (SUM (MAP size_of_shape (dec_shapes nds')))|>
                   nds';
               params = MAP (Var Local ∘ FST) args;
               new_main =
                 Function
                   <|name := start; inline := F; export := F;
                     params := args;
                     body :=
                       Seq (nested_seq decls) (TailCall start' params);
                     return := One|>
             in
               new_main::funs)

           compile ctxt (While e p) =
           While (compile_exp ctxt e) (compile ctxt p)) ∧

           compile ctxt (Call rtyp e es) =

              cexps = MAP (compile_exp ctxt) es

                  (case hdl of
                     NONE => NONE

                  cexps
