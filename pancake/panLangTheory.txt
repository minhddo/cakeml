      ⊢ ∀r. r < 2 ⇔ ∃a. r = varkind2num a
   
   [varkind2num_num2varkind]  Theorem
      
      ⊢ ∀r. r < 2 ⇔ varkind2num (num2varkind r) = r
   
   [varkind2num_thm]  Theorem
      
      ⊢ varkind2num Local = 0 ∧ varkind2num Global = 1
   
   [varkind_Axiom]  Theorem
      
      ⊢ ∀x0 x1. ∃f. f Local = x0 ∧ f Global = x1
   
   [varkind_EQ_varkind]  Theorem
      
      ⊢ ∀a a'. a = a' ⇔ varkind2num a = varkind2num a'
   
   [varkind_case_cong]  Theorem
      
      ⊢ ∀M M' v0 v1.
          M = M' ∧ (M' = Local ⇒ v0 = v0') ∧ (M' = Global ⇒ v1 = v1') ⇒
          (case M of Local => v0 | Global => v1) =
          case M' of Local => v0' | Global => v1'
   
   [varkind_case_def]  Theorem
      
      ⊢ (∀v0 v1. (case Local of Local => v0 | Global => v1) = v0) ∧
        ∀v0 v1. (case Global of Local => v0 | Global => v1) = v1
   
   [varkind_case_eq]  Theorem
      
      ⊢ (case x of Local => v0 | Global => v1) = v ⇔
        x = Local ∧ v0 = v ∨ x = Global ∧ v1 = v
   
   [varkind_distinct]  Theorem
      
      ⊢ Local ≠ Global
   
   [varkind_induction]  Theorem
      
      ⊢ ∀P. P Global ∧ P Local ⇒ ∀a. P a
   
   [varkind_nchotomy]  Theorem
      
      ⊢ ∀a. a = Local ∨ a = Global
