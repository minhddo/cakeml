   [compile_def]  Theorem
      
      ⊢ (∀v s p e ctxt.
           compile ctxt (Dec v s e p) =
           Dec v (compile_shape ctxt.structs s) (compile_exp ctxt e)
             (compile (ctxt with locals := (v,s)::ctxt.locals) p)) ∧
        (∀vk v e ctxt.
           compile ctxt (Assign vk v e) = Assign vk v (compile_exp ctxt e)) ∧
        (∀v ctxt ad.
           compile ctxt (Store ad v) =
           Store (compile_exp ctxt ad) (compile_exp ctxt v)) ∧
        (∀v ctxt ad.
           compile ctxt (Store32 ad v) =
           Store32 (compile_exp ctxt ad) (compile_exp ctxt v)) ∧
        (∀src dest ctxt.
           compile ctxt (StoreByte dest src) =
           StoreByte (compile_exp ctxt dest) (compile_exp ctxt src)) ∧
        (∀p' p ctxt.
           compile ctxt (Seq p p') = Seq (compile ctxt p) (compile ctxt p')) ∧
        (∀p' p e ctxt.
           compile ctxt (If e p p') =
           If (compile_exp ctxt e) (compile ctxt p) (compile ctxt p')) ∧
        (∀p e ctxt.
           compile ctxt (While e p) =
           While (compile_exp ctxt e) (compile ctxt p)) ∧
        (∀rtyp f es ctxt.
           compile ctxt (Call rtyp f es) =
           (let
              cexps = compile_exps ctxt es
            in
              case rtyp of
                NONE => TailCall f cexps
              | SOME (tl,hdl) =>
                Call
                  (SOME
                     (tl,
                      case hdl of
                        NONE => NONE
                      | SOME (eid,evar,p) => SOME (eid,evar,compile ctxt p)))
                  f cexps)) ∧
        (∀v s p f es ctxt.
           compile ctxt (DecCall v s f es p) =
           DecCall v (compile_shape ctxt.structs s) f
             (compile_exps ctxt es)
             (compile (ctxt with locals := (v,s)::ctxt.locals) p)) ∧
        (∀ptr2 ptr1 len2 len1 f ctxt.
           compile ctxt (ExtCall f ptr1 len1 ptr2 len2) =
           ExtCall f (compile_exp ctxt ptr1) (compile_exp ctxt len1)
             (compile_exp ctxt ptr2) (compile_exp ctxt len2)) ∧
        (∀rt ctxt. compile ctxt (Return rt) = Return (compile_exp ctxt rt)) ∧
        (∀excp eid ctxt.
           compile ctxt (Raise eid excp) =
           Raise eid (compile_exp ctxt excp)) ∧
        (∀r op ctxt ad.
           compile ctxt (ShMemStore op r ad) =
           ShMemStore op (compile_exp ctxt r) (compile_exp ctxt ad)) ∧
        (∀vk r op ctxt ad.
           compile ctxt (ShMemLoad op vk r ad) =
           ShMemLoad op vk r (compile_exp ctxt ad)) ∧
        (∀v0. compile v0 Skip = Skip) ∧
        (∀v14 v13 v12 v0.
           compile v0 (Primitive v12 v13 v14) = Primitive v12 v13 v14) ∧
        (∀v0. compile v0 Break = Break) ∧
        (∀v0. compile v0 Continue = Continue) ∧
        (∀v0. compile v0 Tick = Tick) ∧
        ∀v52 v51 v0. compile v0 (Annot v51 v52) = Annot v51 v52
